    notes = get_notes()
    Rref, Rmax = 3., 3.4

    R = np.hypot(notes['sampXYZ'][:,0],notes['sampXYZ'][:,2])/1000.
    Z = notes['sampXYZ'][:,1]/1000.
    R = R-np.min(R[np.abs(Z)<0.01])+Rref
    R[2:4] = Rmax

    Poly1 = np.array([R,Z])
    ind = notes['indedges']
    Ptemp = 0.5*(Poly1[:,ind[0,:]] + Poly1[:,ind[1,:]])
    Poly0 = np.concatenate((Poly1[:,:ind[0,0]], Ptemp,
                            Poly1[:,ind[1,-1]+1:]),axis=1)
